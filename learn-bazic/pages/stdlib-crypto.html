Stdlib ? crypto

std/crypto

Hashing and secure tokens.

SHA-256

let hex = crypto_sha256_hex("hello");
println(hex);

Random tokens

let token = crypto_random_hex(16);
if token.is_ok { println(token.value); }

Passwords

let hash = crypto_bcrypt_hash("secret", 10);
let ok = crypto_bcrypt_verify("secret", hash.value);
