std/json

Validate and transform JSON strings.

Validate

let ok = json_validate("{"a":1}");
println(str(ok));

Pretty print

let res = json_pretty("{"a":1}");
if res.is_ok { println(res.value); }

Read values

let name = json_get_string("{"name":"Ada"}", "name");
if name.is_ok { println(name.value); }
