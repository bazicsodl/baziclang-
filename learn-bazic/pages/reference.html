Quick Reference

Keep this page open while you code.

Program skeleton

fn main(): void {
    println("Hello");
}

Variables

let count: int = 3;
let name = "Ada"; // inferred type

Functions

fn add(a: int, b: int): int {
    return a + b;
}

Structs

struct User { name: string; age: int; }
let u = User{ name: "Sam", age: 19 };

Enums + match

enum Status { Ok, Bad }
let s = Status.Ok;
match s {
    Ok: { println("ok"); }
    Bad: { println("bad"); }
}

Match expression

let label = match s {
    Ok: "OK",
    Bad: "BAD",
};

Loops

let i = 0;
while i < 3 {
    println(str(i));
    i = i + 1;
}

Imports

import "std";
import "./helpers";

Option + Result

let res = parse_int("42");
if res.is_ok { println(str(res.value)); }

Run, build, test

.\bin\bazic.exe run
.\bin\bazic.exe build -o .\bin\app.exe
.\bin\bazic.exe test .
.\bin\bazic.exe lint .
