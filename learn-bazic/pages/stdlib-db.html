std/db

SQLite by default. Drivers available via db_*_with.

SQLite create and query

let ok = db_exec("app.db", "create table if not exists users(id int, name string);");
if !ok.is_ok { println(ok.err.message); return; }
let res = db_query("app.db", "select id, name from users;");
if res.is_ok { println(res.value); }

Parameterized queries

let params = "1	Ada"; // TSV params
let ok = db_exec_params("app.db", "insert into users values(?, ?)", params);

Other drivers

let dsn = "postgres://user:pass@localhost:5432/app?sslmode=disable";
let res = db_query_with("postgres", dsn, "select 1");
