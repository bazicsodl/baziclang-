UI (WASM)

Bazic UI is a minimal JSON UI DSL rendered in the browser.

Create a UI app

.\bin\bazic.exe ui init --dir .\my-ui
.\bin\bazic.exe ui build --dir .\my-ui
.\bin\bazic.exe ui dev --dir .\my-ui --port 8080

Render a layout

import "std";

fn main(): void {
    let nav = ui_nav_link("Home", "nav-home", true);
    let body = ui_p("Hello from Bazic UI");
    let root = ui_layout("Bazic UI", nav, body);
    let _ = ui_render(root);
}

State and events

let clicks = ui_state_int_get("clicks", 0);
let ev = ui_event_type();
if ev == "click" { ui_state_int_set("clicks", clicks + 1); }
ui_event_clear();

Routing

let page = ui_route_get("home");
if page == "settings" { /* render settings */ }
