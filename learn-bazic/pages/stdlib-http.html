std/http

HTTP client and server helpers.

GET a URL

let res = http_get("https://example.com");
if res.is_ok { println(res.value); }

POST JSON

let opts = HttpOptions{
    connect_timeout_ms: 2000,
    timeout_ms: 8000,
    headers: "Accept: application/json",
    user_agent: "BazicClient/1.0",
    content_type: "application/json",
    tls_insecure_skip_verify: false,
    tls_ca_bundle_pem: "",
};
let res = http_post_opts("https://example.com", "{"ok":true}", opts);

Serve text

let ok = http_serve_text(":8080", "hello");
