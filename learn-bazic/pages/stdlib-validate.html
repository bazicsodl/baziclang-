std/validate

Input validation helpers.

Required and length

let ok = validate_required(name, "name");
let ok2 = validate_min_len(name, 3, "name");

Numbers and enums

let ok = validate_int_range(age, 0, 120, "age");
let ok2 = validate_enum(role, "Admin,User,Guest");
