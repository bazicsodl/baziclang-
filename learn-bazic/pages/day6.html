Day 6 ? Strings and files

Goal: read and write text from disk.

Read a file

import "std";

fn main(): void {
    let res = fs_read_file("notes.txt");
    if res.is_ok {
        println(res.value);
    } else {
        println(res.err.message);
    }
}

Write a file

import "std";

fn main(): void {
    let ok = fs_write_file("notes.txt", "hello from bazic");
    if ok.is_ok {
        println("saved");
    } else {
        println(ok.err.message);
    }
}
Try it: Save your name and then read it back.
Checkpoint: You can safely handle errors with Result.
